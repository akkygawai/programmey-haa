PROGRAM ARRAY ORDER REVERSAL

#include<stdio.h>
#include<conio.h>
void main()
{
    int a[50],n,i,j,temp;
    printf("\n ENTER THE ARRAY LIMIT :");
    scanf("%d",&n);
    printf("\n ENTER %d ELEMENTS INTO THE ARRAY :",n);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
        printf("\n BEFORE REVERSAL :\n");
        for(i=0;i<n;i++)
            printf("%d \t",a[i]);
        for(i=0,j=n-1;i<j;i++,j--)
        {
            temp=a[i];
            a[i]=a[j];
            a[j]=temp;
        }
        printf("\n AFTER REVERSAL :\n");
    for(i=0;i<n;i++)
        printf("%d \t",a[i]);
    getch();
}
