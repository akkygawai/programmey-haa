PROGRAM FINDING THE MAXIMUM NUMBER IN A SET

#include <stdio.h>
void main()
 {
 int n,maximumNo;
 int arr[100];
 printf("Enter the number of elements (1 to 100): ");
 scanf("%d", &n);
 for (int i = 0; i < n; ++i)
    {
 printf("Enter number%d: ", i + 1);
 scanf("%d", &arr[i]);
 }
 maximumNo=arr[0];
 for (int i = 1; i < n; ++i)
 {
 if (maximumNo < arr[i])
  {
 maximumNo = arr[i];
 }
 }
 printf("Largest element = %d", maximumNo);
 getch();
}
