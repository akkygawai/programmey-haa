#include<stdio.h>
#include<conio.h>
int main()
{
    int a[50],n,k,i,j,temp,small;
    printf("\n ENTER THE SIZE OF THE ARRAY:");
    scanf("%d",&n);
    printf("\n ENTER THE %d ELEMENTS:",n);

for(i=0;i<n;i++)
scanf("%d",&a[i]);

printf("enter kth value:");
scanf("%d",&k);

    for(i=0;i<n-1;i++)
    {
        for(j=i+1;j<n;j++)
        {
            if(a[i]>a[j])
            {
                temp=a[i];
                a[i]=a[j];
                a[j]=temp;
            }
        }
    }
    small=a[k-1];
    printf("\n kth SMALLEST ELEMENT=%d",small);
    return 0;
}
