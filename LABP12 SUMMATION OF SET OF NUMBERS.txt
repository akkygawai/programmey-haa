PROGRAM SUMMATION OF SET OF NUMBERS

#include<stdio.h>
void main()
{
 int arr[100], size, i, sum = 0;
 printf("Enter the size\n");
 scanf("%d",&size);
 printf("Enter numbers\n");
 for(i = 0; i < size; i++)
 scanf("%d",&arr[i]);
 for(i = 0; i < size; i++)
 sum = sum + arr[i];
 printf("Sum of the entered numbers = %d\n",sum);
 getch();
}
